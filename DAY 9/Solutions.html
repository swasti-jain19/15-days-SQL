Solution 1 a: Creating the songs table with the default value for genre and constraints.

		CREATE TABLE songs (
			song_name VARCHAR(255) NOT NULL,
			genre VARCHAR(100) DEFAULT 'Not defined',
			price DECIMAL(10, 2) CHECK (price >= 1.99) DEFAULT 1.99,
			release_date DATE CHECK (release_date BETWEEN '1950-01-01' AND CURRENT_DATE),
			CONSTRAINT date_check CHECK (release_date BETWEEN '1950-01-01' AND CURRENT_DATE)
        );


Solution 1 b: Inserting a row into the table:

			INSERT INTO songs (song_name, genre, price, release_date)
				VALUES ('Song A', 'Pop', 2.99, '2024-11-28');


Solution 1 c: Modifying the constraint to allow a price of 0.99:

			ALTER TABLE songs
				DROP CONSTRAINT price_check;

				ADD CONSTRAINT price_check CHECK (price >= 0.99);
